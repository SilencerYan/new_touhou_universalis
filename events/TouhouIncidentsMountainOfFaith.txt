#Mountain of faith Incident - Set up the incident
country_event = {
	id = th_mountain_of_faith_incident_events.0
	title = "th_mountain_of_faith_incident_events.0.t"
	desc = "th_mountain_of_faith_incident_events.0.desc"
	picture = MOUNTAINOFFAITH_eventPicture

	trigger = {
		th_incident_check_year_for_incident = { incident = mountain_of_faith_incident year = 1525 }
		NOT = { th_incident_has_been_resolved = { incident = mountain_of_faith_incident } }
		th_incident_can_trigger_generic_check = yes
		OR = { 		#MRY must be moriya shintoism or not existing
			NOT = { exists = MRY }
			MRY = { religion = th_moriyashinto }
		}
	}
	
	fire_only_once = yes
	hidden = yes	
	
	mean_time_to_happen = {
		months = 12

		modifier = {
			factor = 0.8
			exists = MRY
		}
		modifier = {
			factor = 0.5
			MRY = {
				is_at_war = yes
			}
		}
		modifier = {
			factor = 0.5
			OR = {
				AND = {
					tag = MRY
					NOT = { exists = HKR }
				}
				tag = FOM
				tag = SCR
				tag = HKR
				AND = {
					tag = GNK
					has_reform = Gensokyan_shrine_theocracy
					NOT = { exists = HKR }
					NOT = { exists = MRY }
				}
			}
		}
		modifier = {
			factor = 0.2
			AND = {
				OR = {
					tag = SCR
					tag = SCD
				}
				ai = no
			}
		}
		modifier = {
			factor = 0.15
			AND = {
				tag = FOM
				ai = no
			}
		}
		modifier = {
			factor = 0.1
			AND = {
				tag = HKR
				ai = no
			}
		}
	}

	immediate = {
		th_incident_forbid_country_to_investigate = { who = MRY }
		th_incident_forbid_country_to_investigate = { who = AKI }
		th_incident_forbid_country_to_investigate = { who = YOF }
		th_incident_forbid_country_to_investigate = { who = KPP }
		th_incident_forbid_country_to_investigate = { who = TGU }
		th_start_incident_effect = {
			incident = mountain_of_faith_incident
		}
		every_country = {
			#NO exceptions-even MRY itself
			limit = {
				capital_scope = {
					superregion = gensokyo_superregion
				}
			}
			#Every Gensokyan country get tolerance of heretic -4 debuff
			add_country_modifier = {
				name = th_MOF_religious_turmoil_negative
				duration = -1
				desc = th_incident_duration_tt
			}
		}
	}

	option = {		#Check any possible investigator country
		random_country = {
			limit = { 
				th_incident_can_become_investigator = yes
			}
			th_incident_setup_investigator = yes
			th_incident_introduction_event = {
				id = th_mountain_of_faith_incident_events.1
			}
		}
		ai_chance = { 
			factor = 100
		}
	}
}
