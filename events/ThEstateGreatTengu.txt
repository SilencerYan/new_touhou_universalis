#Great Tengu encroach on Government Lands
country_event = {
	id = th_estate_great_tengu_events.4
	title = th_estate_great_tengu_events.4.t
	desc = th_estate_great_tengu_events.4.desc
	picture = GREAT_TENGU_GOOD_eventPicture
	goto = noble_encroachment
	
	trigger = {
		has_estate = estate_great_tengu
		num_of_cities = 15
		any_owned_province = {
			is_overseas = no
			is_capital = no
			is_colony = no
			has_seat_in_parliament = no
			is_territory = no
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					is_overseas = no
					is_colony = no
					has_seat_in_parliament = no
					is_territory = no
				}
				save_event_target_as = noble_encroachment
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_great_tengu
				influence = 80
			}
			NOT = {
				estate_loyalty = {
					estate = estate_great_tengu
					loyalty = 80
				}
			}
		}
	}
	
	option = {
		name = th_estate_great_tengu_events.4.a #Allow it.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				estate_territory = {
					estate = estate_great_tengu
					territory = 12
				}
			}
			modifier = {
				factor = 0.25
				estate_influence = {
					estate = estate_great_tengu
					influence = 60
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_great_tengu
						loyalty = 30
					}
				}
			}
		}
		event_target:noble_encroachment = {
			add_local_autonomy = 25
		}
		give_estate_land_share_small = { estate = estate_great_tengu }
	}
	option = {
		name = th_estate_great_tengu_events.4.b #Take it Back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.1
				NOT = {
					estate_territory = {
						estate = estate_great_tengu
						territory = 15
					}
				}
			}
			modifier = {
				factor = 0.25
				NOT = {
					estate_loyalty = {
						estate = estate_great_tengu
						loyalty = 35
					}
				}
			}
		}
		event_target:noble_encroachment = {
			random_list = {
				50 = {
					spawn_rebels = {
						type = th_great_tengu_rebels
						size = 1
					}
				}
				50 = {  }
			}
		}
		add_estate_loyalty = {
			estate = estate_great_tengu
			loyalty = -15
		}
	}
}
