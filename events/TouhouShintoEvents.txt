			factor = 100
		}

	}

			NOT = { owns = 4955 }
		}

			limit = {
				owns = 4955
			}
			country_event = {
				id = TH_shinto_events.8
				days = 15
			}
			add_opinion = {
				who = ROOT
				modifier = th_shinto_visited_shrine
			}
		}

		add_dip_power = 25
		add_prestige = 10
		add_treasury = -100
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}

		ai_chance = { 
			factor = 35
			modifier = {
				factor = 0
				NOT = { treasury = 100 }
			}
			modifier = {
				factor = 0
				4955 = {
					owner = {
						NOT = {
							reverse_has_opinion = {
								who = ROOT
								value = 100
							}
						}
