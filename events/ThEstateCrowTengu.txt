				limit = {

				}
			}
		}
		add_estate_influence_modifier = {
			estate = estate_crow_tengu
			desc = EST_VAL_CROW_TENGU_HALTED_COMPETITIONS
			influence = -15
			duration = 5475
		}
		every_owned_province = {
			limit = { has_province_flag = th_competition_revolt }
			clr_province_flag = th_competition_revolt
			spawn_rebels = {
				type = th_crow_tengu_rebels
				size = 1
			}
		}
		add_estate_loyalty = {
			estate = estate_crow_tengu

		}

			influence = 60
		}
		estate_loyalty = {
			estate = estate_crow_tengu
			loyalty = 60
		}
		NOT = {
			any_active_trade_node = {
				has_merchant = prev
				has_trade_modifier = {
					who = ROOT
					key = STRONG_MERCHANTS
				}
			}
		}
		any_owned_province = {
			OR = {
				owner = { government = republic }
				development = 10
				province_trade_power = 5
			}
			is_capital = no
			is_city = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_list = {
				65 = { set_country_flag = growing_trade_influence }
				35 = { set_country_flag = growing_cities }
			}
		}
	}
	
	#Let them rest
	option = {
		name = th_estate_crow_tengu_events.6.a
		trigger_switch = {
			on_trigger = has_country_flag
			growing_trade_influence = {
				random_owned_province = {
					limit = {
						OR = {
							owner = { government = republic }
							development = 10
							province_trade_power = 5
						}
						is_capital = no
						is_city = yes
					}
					add_trade_modifier = {
						who = root
						duration = 5475
						power = 15
						key = STRONG_MERCHANTS
					}
				}
				clr_country_flag = growing_trade_influence
			}
			growing_cities = {
				random_owned_province = {
					limit = {
						OR = {
							owner = { government = republic }
							development = 10
							province_trade_power = 5
						}
						is_city = yes
					}
					add_base_production = 1
				}
				clr_country_flag = growing_cities
			}
		}
	}
}
