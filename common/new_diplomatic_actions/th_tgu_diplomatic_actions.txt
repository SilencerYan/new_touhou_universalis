#Common Diplo Actions, Tengu states have ALWAYS access to these
th_sell_newspaper = {
	category = th_diplo_action
	require_acceptance = yes

	is_visible = {
		OR = {
			th_has_copied_from_target = {
				target = TGU
			}
			has_government_attribute = enables_tengu_newspaper
		}
	}

	is_allowed = {
		if = {
			limit = {
				FROM = { home_trade_node = { NOT = { trade_range = ROOT } } }
			}
			custom_trigger_tooltip = {
				tooltip = TH_TARGETS_HOME_TRADE_NODE_OUT_OF_REACH_tt
				FROM = { home_trade_node = { trade_range = ROOT } }
			}
		}
		if = {
			limit = {
				is_rival = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_IS_RIVAL
				NOT = { is_rival = FROM }
			}
		}
		if = {
			limit = {
				is_enemy = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_IS_ENEMY
				NOT = { is_enemy = FROM }
			}
		}
		if = {
			limit = {
				war_with = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_CANNOT_USE_AGAINST_WAR_ENEMIES_TT
				NOT = { war_with = FROM }
			}
		}
		if = {
			limit = {
				FROM = {
					has_country_flag = th_received_newspaper_from_@ROOT
					NOT = {
						had_country_flag = {
							flag = th_received_newspaper_from_@ROOT
							days = 3650
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_CAN_SELL_ONLY_EVERY_10_YEARS_TT
				FROM = {
					OR = {
						NOT = { has_country_flag = th_received_newspaper_from_@ROOT }
						had_country_flag = {
							flag = th_received_newspaper_from_@ROOT
							days = 3650
						}
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { is_subject_other_than_tributary_trigger = yes }
			}
			custom_trigger_tooltip = {
				tooltip = TH_CANNOT_DO_ACTION_TO_SUBJECTS_TT
				FROM = { is_free_or_tributary_trigger = yes }
			}
		}
		if = {
			limit = {
				NOT = {
					any_owned_province = {
						trade_goods = paper
						NOT = { has_province_modifier = th_is_selling_exclusive_newspaper }
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_NEEDS_ONE_AVAILABLE_PAPER_PROVINCE_TT
				any_owned_province = {
					trade_goods = paper
					NOT = { has_province_modifier = th_is_selling_exclusive_newspaper }
				}
			}
		}
		if = {
			limit = {
				trade_embargoing = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_FROM
				NOT = { trade_embargoing = FROM }
			}
		}
		if = {
			limit = {
				trade_embargo_by = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_BY_FROM
				NOT = { trade_embargo_by = FROM }
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		change_variable = {
			which = tgu_newspaper_sold_counter
			value = 1
		}
		FROM = {
			clr_country_flag = th_received_newspaper_from_@ROOT
			set_country_flag = th_received_newspaper_from_@ROOT
			country_event = {
				id = th_diplo_events.51
			}
		}
		random_owned_province = {
			limit = {
				trade_goods = paper
				NOT = { has_province_modifier = th_is_selling_exclusive_newspaper }
			}
			add_province_modifier = {
				name = th_is_selling_exclusive_newspaper
				duration = 3650
			}
		}
	}

    ai_acceptance = {
        add_entry = {
            name = TH_DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = TH_DIPLO_TRUST_WITH
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_hostile
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -50
			}
		}
		add_entry = {
			name = DIPLO_WANT_WAR
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_outraged
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value

				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			
			multiply_variable = {
				which = ai_value
				value = -1000
			}
		}
		add_entry = {
			name = DIPLO_FRIENDLINESS
			limit = {

th_spread_good_image = {

	is_allowed = {
		if = {

				FROM = { home_trade_node = { trade_range = ROOT } }
			}
		}
		if = {
			limit = {
				NOT = {
					has_spy_network_in = {
						who = FROM
						value = 30
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_REQUIRES_30_SPY_NETWORK_TT
				has_spy_network_in = {
					who = FROM
					value = 30
				}
			}
		}
		if = {
			limit = {
				FROM = {
					has_country_flag = th_received_good_image_of_@ROOT
					NOT = {
						had_country_flag = {
							flag = th_received_good_image_of_@ROOT
							days = 3650
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_spread_good_image_cd_tt
				FROM = {
					OR = {
						NOT = { has_country_flag = th_received_good_image_of_@ROOT }
						had_country_flag = {
							flag = th_received_good_image_of_@ROOT
							days = 3650
						}
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
		if = {
			limit = {
				trade_embargoing = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_FROM
				NOT = { trade_embargoing = FROM }
			}
		}
		if = {
			limit = {
				trade_embargo_by = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_BY_FROM
				NOT = { trade_embargo_by = FROM }
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		change_variable = {
			which = tgu_newspaper_action_counter
			value = 1
		}
		if = {
			limit = { NOT = { has_country_flag = tgu_spread_goods_image } }
			set_country_flag = tgu_spread_goods_image
		}
		FROM = {
			clr_country_flag = th_received_good_image_of_@ROOT
			set_country_flag = th_received_good_image_of_@ROOT
			if = {
				limit = {
					has_country_flag = th_was_spread_good_image_target_once
				}
				clr_country_flag = th_was_spread_good_image_target_once
				set_country_flag = th_was_spread_good_image_target_twice
			}
			else = {
				set_country_flag = th_was_spread_good_image_target_once
			}
			remove_opinion = {
				who = ROOT
				modifier = caught_spy
			}
			remove_opinion = {
				who = ROOT
				modifier = declared_war_on_neighbor
			}
			add_opinion = {
				who = ROOT
				modifier = th_good_image
			}
		}
		add_spy_network_in = {
			who = FROM
			value = -30
		}
	}

	ai_will_do = {
		always = no
	}
}

th_promote_puppet_to_vassal = {
	category = th_diplo_action

	require_acceptance = no

	is_visible = {
		OR = {
			th_has_copied_from_target = {
				target = TGU
			}
			AND = {
				has_government_attribute = enables_tengu_newspaper
				has_country_flag = tgu_has_extended_puppet_state_actions
			}
		}
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = th_tengu_puppet_state
		}
	}

	is_allowed = {
		if = {
			limit = {
				NOT = {
					has_spy_network_in = {
						who = FROM
						value = 80
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_REQUIRES_80_SPY_NETWORK_TT
				has_spy_network_in = {
					who = FROM
					value = 80
				}
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = {
						trust = {
							who = ROOT
							value = 75
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_HAS_TRUST_LEVEL_75_TT
				FROM = {
					trust = {
						who = ROOT
						value = 75
					}
				}
			}
		}
		if = {
			limit = {
				is_at_war = yes
			}
			custom_trigger_tooltip = {
				tooltip = TH_IS_NOT_AT_WAR_TT
				is_at_war = no
			}
		}
		if = {
			limit = {
				FROM = { is_at_war = yes }
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_IS_NOT_AT_WAR_TT
				FROM = { is_at_war = no }
			}
		}
		if = {
			limit = {
				FROM = { total_development = 200 }
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_TOTAL_DEV_NOT_200_TT
				FROM = { NOT = { total_development = 200 } }
			}
		}
	}

	on_accept = {
		add_spy_network_in = {
			who = FROM
			value = -80
		}
		FROM = { grant_independence = yes }
		create_subject = {
			subject_type = vassal
			subject = FROM
		}
	}

	ai_will_do = {
		always = yes
	}
}

th_denounce_allies = {
	category = th_diplo_action
	require_acceptance = no

	is_visible = {
		OR = {
			th_has_copied_from_target = {
				target = TGU
			}
			AND = {
				has_government_attribute = enables_tengu_newspaper
				has_country_flag = tgu_has_enabled_military_spy_actions
			}
		}
		FROM = { is_free_or_tributary_trigger = yes }
	}

	is_allowed = {
		if = {
			limit = {
				FROM = { home_trade_node = { NOT = { trade_range = ROOT } } }
			}
			custom_trigger_tooltip = {
				tooltip = TH_TARGETS_HOME_TRADE_NODE_OUT_OF_REACH_tt
				FROM = { home_trade_node = { trade_range = ROOT } }
			}
		}
		if = {
			limit = {
				NOT = {
					has_spy_network_in = {
						who = FROM
						value = 60
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_REQUIRES_60_SPY_NETWORK_TT
				has_spy_network_in = {
					who = FROM
					value = 60
				}
			}
		}
		if = {
			limit = {
				FROM = {
					has_country_flag = th_had_to_denounce_allies_by_@ROOT
					NOT = {
						had_country_flag = {
							flag = th_had_to_denounce_allies_by_@ROOT
							days = 1825
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_CAN_DENOUNCE_EVERY_5_YEARS_TT
				FROM = {
					OR = {
						NOT = { has_country_flag = th_had_to_denounce_allies_by_@ROOT }
						had_country_flag = {
							flag = th_had_to_denounce_allies_by_@ROOT
							days = 1825
						}
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						has_country_flag = th_immunity_to_diplomatic_actions
						has_government_attribute = enables_yama_authority
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = th_from_is_not_immune_to_diplomatic_actions_tt
				FROM = {
					NOT = { has_country_flag = th_immunity_to_diplomatic_actions }
					NOT = { has_government_attribute = enables_yama_authority }
				}
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = { num_of_allies = 1 }
				}
			}
			custom_trigger_tooltip = {
				tooltip = TH_FROM_NEEDS_TO_HAVE_ALLIES
				FROM = {
					num_of_allies = 1
				}
			}
		}
		if = {
			limit = {
				trade_embargoing = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_FROM
				NOT = { trade_embargoing = FROM }
			}
		}
		if = {
			limit = {
				trade_embargo_by = FROM
			}
			custom_trigger_tooltip = {
				tooltip = TH_EMBARGOING_BY_FROM
				NOT = { trade_embargo_by = FROM }
			}
		}
	}

	on_accept = {
		th_clear_copy_flag = yes
		FROM = {
			clr_country_flag = th_had_to_denounce_allies_by_@ROOT
			set_country_flag = th_had_to_denounce_allies_by_@ROOT
			every_ally = {
				reverse_add_opinion = {
					who = FROM
					modifier = th_ally_denouncement
				}
			}
		}
		add_spy_network_in = {
			who = FROM
			value = -60
		}
	}

		always = no
	}
