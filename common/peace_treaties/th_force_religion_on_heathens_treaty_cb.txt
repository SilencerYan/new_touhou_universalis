#Force Religion on Heathens - only usable through special cbs

po_th_force_religion_cb = {
	category = 6
	power_projection = vassalized_rival
	power_cost_base = 1.0
	prestige_base = 1
	ae_base = 0
	warscore_cost = {
		all_provinces = 0.0
		no_provinces = 50.0
		owner_keeps = 0.0
		conquered = 0.0
		returned_core = 0.0
		released_nation = 0.0
		cancelled_subject = 0.0
		concede_colonial_region = 0.0
	}
	warscore_cap = -1
	requires_demand_independence = yes
	is_make_subject = no
	requires_is_allowed = yes
	
	is_visible = {
		NOT = { religion = FROM }
	}
	is_allowed = {
		FROM = {
			NOT = { tag = PAP }
			NOT = { religion = THIS }
		}
	}
	effect = {
		FROM = { 
			change_religion = ROOT
			force_converted = yes
		}
	}
	ai_weight = { # Defaults to always evaluate to 0
		export_to_variable = {
			variable_name = ai_value # Mandatory to have entry called "ai_value". Its value at the end of the ai_weight scope is what will be used
			value = 10
		}
		limit = {
			always = yes
		}
	}
}
