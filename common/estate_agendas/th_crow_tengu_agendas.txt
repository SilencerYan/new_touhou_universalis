estate_crow_tengu_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
	}
	selection_weight = {
		factor = 10
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_crow_tengu
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_crow_tengu
			}
		}
		else = {
			employed_advisor = {
				category = MIL
			}
			is_in_deficit = no
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_crow_tengu
		}
		add_estate_loyalty = {
			estate = estate_crow_tengu
			loyalty = 10
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		th_on_failed_agenda_effect = yes
        add_estate_loyalty_modifier = {
            estate = estate_crow_tengu
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Recruit more guards
estate_crow_tengu_recruit_more_cossacks = {
	can_select = {
		has_estate_privilege = estate_crow_tengu_establish_tenma_vanguards
		OR = {
			NOT = { cossacks_percentage = 0.1 }
			AND = {
				has_estate_privilege = estate_crow_tengu_expand_the_tenma_vanguards
				NOT = { cossacks_percentage = 0.2 }
			}
		}
	}
	selection_weight = {
		factor = 15
	}
	pre_effect = {
		set_variable = {
			which = estate_crow_tengu_recruit_more_cossacks_var
			value = 0
		}
		if = {
			limit = {
				cossacks_percentage = 0.15
			}
			set_variable = {
				which = estate_crow_tengu_recruit_more_cossacks_var
				value = 4
			}
		}
		else_if = {
			limit = {
				cossacks_percentage = 0.1
			}
			set_variable = {
				which = estate_crow_tengu_recruit_more_cossacks_var
				value = 3
			}
		}
		else_if = {
			limit = {
				cossacks_percentage = 0.05
			}
			set_variable = {
				which = estate_crow_tengu_recruit_more_cossacks_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_crow_tengu_recruit_more_cossacks_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_crow_tengu
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_crow_tengu
			}
		}
		else = {
			if = {
				limit = {
					check_variable = {
						which = estate_crow_tengu_recruit_more_cossacks_var
						value = 4
					}
				}
				cossacks_percentage = 0.2
			}
			else_if = {
				limit = {
					check_variable = {
						which = estate_crow_tengu_recruit_more_cossacks_var
						value = 3
					}
				}
				cossacks_percentage = 0.15
			}
			else_if = {
				limit = {
					check_variable = {
						which = estate_crow_tengu_recruit_more_cossacks_var
						value = 2
					}
				}
				cossacks_percentage = 0.1
			}
			else = {
				cossacks_percentage = 0.05
			}
		}
	}
	fail_if = {
		NOT = { has_estate_privilege = estate_crow_tengu_establish_tenma_vanguards }
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_crow_tengu
		}
		add_estate_loyalty = {
			estate = estate_crow_tengu
			loyalty = 10
		}
		add_army_tradition = 10
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		th_on_failed_agenda_effect = yes
        add_estate_loyalty_modifier = {
            estate = estate_crow_tengu
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
